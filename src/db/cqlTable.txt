CREATE TABLE messages (
  convoID UUID,
  bucket INT,
  messageID TIMEUUID,

  senderID UUID,          -- NULL for system messages
  content TEXT,           -- Optional for system messages

  messageType TEXT,       -- 'text' | 'image' | 'video' | 'system'
  systemType TEXT,        -- 'user_joined' | 'user_left' | 'user_added' | 'user_kicked'

  actorId UUID,           -- who performed the action
  targetUserId UUID,      -- who was affected

  attachments LIST<TEXT>,
  isEdited BOOLEAN,
  editedAt TIMESTAMP,
  isDeleted BOOLEAN,
  deletedAt TIMESTAMP,
  replyToMessageID TIMEUUID,

  PRIMARY KEY ((convoID, bucket), messageID)
)
WITH CLUSTERING ORDER BY (messageID DESC);



CREATE TABLE message_reads (
  convoID UUID,
  messageID TIMEUUID,
  userID UUID,
  readAt TIMESTAMP,
  PRIMARY KEY ((convoID, messageID), userID)
) WITH COMPRESSION = {'class': 'org.apache.cassandra.io.compress.LZ4Compressor'};


CREATE TABLE message_reactions (
  convoID UUID,
  messageID TIMEUUID,
  userID UUID,
  reaction TEXT,
  reactedAt TIMESTAMP,
  PRIMARY KEY ((convoID, messageID), userID)
);
